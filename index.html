// Update product count display function updateProductCount(total, filtered) { const countNum = document.getElementById('countNum'); const productCount = document.getElementById('productCount'); if (countNum) { // Animate the number change const currentCount = parseInt(countNum.textContent) || 0; const targetCount = total || 0; animateCount(countNum, currentCount, targetCount); } // Update stat cards document.getElementById('sTotal').textContent = total ? total.toLocaleString() : '–'; document.getElementById('sFiltered').textContent = filtered ? filtered.toLocaleString() : '–'; } // Smooth number animation function animateCount(element, start, end) { const duration = 800; const startTime = performance.now(); function update(currentTime) { const elapsed = currentTime - startTime; const progress = Math.min(elapsed / duration, 1); const eased = 1 - Math.pow(1 - progress, 3); // ease-out const current = Math.floor(start + (end - start) * eased); element.textContent = current.toLocaleString(); if (progress < 1) { requestAnimationFrame(update); } } requestAnimationFrame(update); } // Call this when data loads: // updateProductCount(totalProducts, filteredProducts);
